import Foundation

class NotesApp {
    private var notes: [String] = []

    func addNote(_ note: String) {
        notes.append(note)
        print("Note added: \(note)")
    }

    func deleteNote(at index: Int) {
        if index >= 0 && index < notes.count {
            let removed = notes.remove(at: index)
            print("Note deleted: \(removed)")
        } else {
            print("Invalid index")
        }
    }

    func listNotes() {
        print("\nYour Notes:")
        for (index, note) in notes.enumerated() {
            print("\(index + 1). \(note)")
        }
    }
}

// Simulation
let myNotesApp = NotesApp()

myNotesApp.addNote("Prepare resume for iOS job")
myNotesApp.addNote("Complete SwiftFiddle project")
myNotesApp.addNote("Submit by 11 AM tomorrow")

myNotesApp.listNotes()

myNotesApp.deleteNote(at: 1)
myNotesApp.listNotes()
